```javascript
function hideElements(elementId) {
    const element = document.getElementById(elementId);
    
    if(element.children.length === 0) {
        element.innerHTML = '&#128504;' + element.textContent;
    }
    
    const similarElements = document.getElementsByTagName(element.nodeName);
    
    similarElements.forEach(e => {
        const rightNeighbor = e.nextElementSibling; 
        if(rightNeighbor) {
            rightNeighbor.hidden = true;
        }
    });
}
```

const getInfoMethod = function() {
    const parent = this.parentElement.nodeName;
    const leftNeighbor = this.previousElementSibling?.nodeName ?? 'нет';
    const rightNeighbor = this.nextElementSibling?.nodeName ?? 'нет';
    const firstChild = this.firstElementChild?.nodeName;
    const lastChild = this.lastElementChild?.nodeName;
    const childrenCount = this.children.length;

    return `Родитель: ${parent}\n` +
            `Соседи: слева - ${leftNeighbor}, справа - ${rightNeighbor}\n` +
            `Количество детей: ${childrenCount}` +
            (childrenCount > 0 ? `\nДети: первый - ${firstChild}, последний - ${lastChild}` : '');
}

function setNewProperty(tag) {     
    document.getElementsByTagName(tag).forEach(e => e.getInfo = getInfoMethod);
}

## Задание 12
```javascript
function changeAttribute(hrefDict) {
    const elementsList = document.getElementsByTagName('a')
    
    for(const element of elementsList) {
        for(const hrefKey in hrefDict) {
            if(element.innerHTML.includes(hrefKey)) {
                element.setAttribute('href', hrefDict[hrefKey]);
            }
        }
    }
}
```

```javascript
function setClass(attrName, className, element = document.body) {
    for(const child of element.children) {
        if(child.hasAttribute(attrName)) {
            child.removeAttribute(attrName);
            child.classList.add(className);
        }
        
        setClass(attrName, className, child);
    }
}
```
