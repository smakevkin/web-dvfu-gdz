```javascript
function createCities(arr) {
    let cityes = [];
    
    for(let i = 0; i < arr.length; i++) {
        let char = arr[i];
        
        let cityesIndex = char.indexOf("||");
        let cityIndex = char.slice(cityesIndex + 2);
        let lowerCityes = cityIndex.toLowerCase();
        let resultCityes = lowerCityes[0].toUpperCase() + lowerCityes.slice(1);
        
        cityes.push(resultCityes);
    };
    
    let cityObj = {};
    for (let city of cityes) {
        if (cityObj[city]) {
            cityObj[city]++;
        } else {
            cityObj[city] = 1;
        };
    };
    
    return cityObj;
};
```

```javascript
function createArrBonusStudents(students) {
    const bonusedStudents = [];
    
    for(const student of students) {
        const bonus = student.rate.split(' ').reduce((min, e) => Math.min(min, +e), 5);
        
        if(bonus < 4) {
            continue;
        }
        
        const bonusedStudent = {};
        
        for(const key in student) {
            bonusedStudent[key] = student[key];
        }
        
        bonusedStudent.bonus = bonus;
        bonusedStudents.push(bonusedStudent);
    }
    
    if(bonusedStudents.length === 0) {
        return 'Студентов для начисления бонусов нет';
    }
    
    return bonusedStudents;
}
```

```javascript
function capitalize(str) {
    return str.slice(0, 1).toUpperCase() + str.slice(1).toLowerCase();
}

function userCities(usersArray) {
    let sortedUsersArray = [];
    
    for(const str of usersArray) {
        const userName = capitalize(str.split(':')[0]);
        let userCities = str
            .split(': ')[1]
            .split(', ')
            .map(capitalize);
        
        userCities = userCities.filter((city, index) => userCities.indexOf(city) === index);
        
        if(userCities.length <= 2) {
            continue;
        }
        
        userCities.sort();
        sortedUsersArray.push({ 
            name: userName, 
            cities: userCities
        });
    }
    
    sortedUsersArray.sort((u1, u2) => u1.name.localeCompare(u2.name));
    
    const usersObject = {};
    
    for(const user of sortedUsersArray) {
        usersObject[user.name] = user.cities;
    }
    
    return usersObject;
}
```
