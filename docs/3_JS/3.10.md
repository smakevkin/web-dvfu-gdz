function addNumbers() {
    const numbers = this.innerHTML
        .split(ellipsis)[0]
        .split(' ')
        .map(s => +s);

function getLeftTop(eventX, eventY, rect, size) {
    eventX -= Math.max(0, (eventX + size + 1) - rect.right);
    eventY -= Math.max(0, (eventY + size + 1) - rect.bottom);
    eventX += Math.max(0, rect.left - eventX + 1);
    eventY += Math.max(0, rect.top - eventY + 1);
    return [eventX, eventY]
}
```

function getElements(str, one, two) {
    let numbers = str.split(':')[1].trim().split(' ').map(n => +n);
    numbers.push(+one, +two);
    let arithmetic = numbers.filter((e, i, a) => {
        return i === 0 || a[i] - a[i - 1] == a[1] - a[0];
    });
    let geometric = numbers.filter((e, i, a) => {
        return i === 0 || a[i] * a[0]  == a[i - 1] * a[1];
    });
    let fib = numbers.filter((e, i, a) => {
        return i === 0 || i === 1 || a[i] == a[i - 1] + a[i - 2];
    });
    let digitSum = numbers.filter((e, i, a) => {
        let sum = num => [...String(Math.abs(num))].reduce((sum, d) => sum + +d, 0);
        return i === 0 || a[i] == a[i - 1] + sum(a[i - 1]);
    });
    return numbers.length == geometric.length || numbers.length == arithmetic.length || numbers.length == fib.length || numbers.length == digitSum.length;
}

function validate(username, experience, level) {
    experience = +experience;
    level = +level;
    
    if(!username) {
        return 'Введите имя';
    }
    if(!experience) {
        return 'Введите опыт работы';
    }
    if(experience < 0 || experience > 99) {
        return 'Вы ввели неверное значение опыта работы';
    }
    
    return `${username}, нас устраивает Ваш опыт работы (${experience} ${getYearWord(experience)}) и уровень квалификации (${level}). Приходите на собеседование.`
}

function getYearWord(year) {
    if(5 <= year && year <= 20) {
        return 'лет';
    }
    
    const lastDigit = year % 10;
    
    if(lastDigit === 1) {
        return 'год';
    }
    if(2 <= lastDigit && lastDigit <= 4) {
        return 'года';
    }
    
    return 'лет';
}
```
